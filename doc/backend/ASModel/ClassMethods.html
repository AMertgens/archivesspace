def to_jsonmodel(obj, model)
  if obj.is_a? Integer
    # An ID.  Get the Sequel row for it.
    obj = get_or_die(obj)
  end
  json = JSONModel(model).from_hash(obj.values.reject {|k, v| v.nil? })

  json.uri = json.class.uri_for(obj.id, {:repo_id => obj[:repo_id]})

  json
end
