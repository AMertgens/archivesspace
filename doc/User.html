def self.login(username, password)
  uri = JSONModel(:user).uri_for(username)

  response = JSONModel::HTTP.post_form("#{uri}/login", :password => password)

  if response.code == '200'
    JSON.parse(response.body)["session"]
  else
    nil
  end
end
