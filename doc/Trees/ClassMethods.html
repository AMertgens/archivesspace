def assemble_tree(node, links, properties)
  result = properties[node].clone

  if links[node]
    result['children'] = links[node].sort_by(&:first).map do |position, child_id|
      assemble_tree(child_id, links, properties)
    end
  else
    result['children'] = []
  end

  result
end
