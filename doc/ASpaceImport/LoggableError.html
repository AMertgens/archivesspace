def to_s
  
  s = "#{@header}\n"
  s << "Record type: #{@record_info[:type].capitalize} \n"
  s << "Record title: #{@record_info[:title].capitalize} \n"
  s << "Error messages: "
  s << @messages.join(' : ')
  s
end
