def self.discharge_proxy(record_type, proxied_obj)
  puts "Discharge #{record_type} #{proxied_obj.inspect}"
  if @proxies.has_key?(record_type)
    @proxies[record_type].discharge(proxied_obj)
    @proxies.delete(record_type)
  end
end
