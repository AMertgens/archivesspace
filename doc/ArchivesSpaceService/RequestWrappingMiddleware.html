def call(env)
  start_time = Time.now
  session_token = env["HTTP_X_ARCHIVESSPACE_SESSION"]

  session = nil

  if session_token
    session = Session.find(session_token)

    if session.nil?
      return [412,
              {"Content-Type" => "application/json"},
              [{
                 :code => "SESSION_GONE",
                 :error => "No session found for #{session_token}"
               }.to_json]]
    end
  end

  @app.instance_eval {
    @session = session
  }

  if DB.connected?
    result = DB.open do
      @app.call(env)
    end
  else
    result = @app.call(env)
  end

  end_time = Time.now

  if ArchivesSpaceService.development?
    Log.debug("Responded with #{result} in #{(end_time - start_time) * 1000}ms")
  end

  result
end
