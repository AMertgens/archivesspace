def save_or_wait
  if self.try_save[0]
    while @@wait_queue.length > 0 do
      @@wait_queue.each_index do |i|        
        @@wait_queue[i] = nil if @@wait_queue[i].try_save[0]
      end
      break if @@wait_queue.compact! == nil
    end
  else
    @@wait_queue.push(self)
  end
end
