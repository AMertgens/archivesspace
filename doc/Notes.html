def update_from_json(json, opts = {})
  notes_blob = JSON(json.notes)
  json.notes = nil

  obj = super

  ps = self.class.dataset.where(:id => self.id).prepare(:update, :update_notes, :notes => :$notes)
  ps.call(:notes => notes_blob.to_sequel_blob)

  obj
end
