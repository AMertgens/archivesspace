def ref_id
  ref_id = ""
  
  (0..3).each do |i|
    next if self.send("id_#{i.to_s}").blank?
    ref_id << " - " unless i === 0
    ref_id << self.send("id_#{i.to_s}")
  end
  
  ref_id
end

def tree
  resources = self.linked_records(:spawned).map {|resource|
    {
      :title => resource.title,
      :id => resource.id,
      :node_type => 'resource',
      :record_uri => resource.uri,
    }
  }

  JSONModel(:accession_tree).
      from_hash(:title => self.title,
                :id => self.id,
                :node_type => 'accession',
                :children => resources,
                :record_uri => self.uri)
end
